## Overview

Query historical trade data from Kalshi markets with flexible filtering by ticker, event, time range, and taker side.

<ParamField query="ticker" type="string">
  Filter by market ticker (exact match)
</ParamField>

<ParamField query="event_ticker" type="string">
  Filter by event ticker (prefix match - returns all markets in the event)
</ParamField>

<ParamField query="taker_side" type="string">
  Filter by taker side: `yes` or `no`
</ParamField>

<ParamField query="start_time" type="integer">
  Unix timestamp (seconds) for start of range
</ParamField>

<ParamField query="end_time" type="integer">
  Unix timestamp (seconds) for end of range
</ParamField>

<ParamField query="min_count" type="integer">
  Minimum contract count per trade
</ParamField>

<ParamField query="order" type="string" default="desc">
  Sort order: `asc` or `desc` by timestamp
</ParamField>

<ParamField query="limit" type="integer" default="100">
  Number of trades to return (1-1000)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of trades to skip for pagination (max 100000)
</ParamField>

<Warning>
At least one of `ticker` or `event_ticker` is required.
</Warning>

<ResponseField name="trades" type="array">
  <Expandable title="Trade object">
    <ResponseField name="trade_id" type="string">Unique trade identifier</ResponseField>
    <ResponseField name="ticker" type="string">Market ticker</ResponseField>
    <ResponseField name="count" type="integer">Number of contracts traded</ResponseField>
    <ResponseField name="yes_price" type="number">Yes price (0-1)</ResponseField>
    <ResponseField name="no_price" type="number">No price (0-1)</ResponseField>
    <ResponseField name="taker_side" type="string">`yes` or `no`</ResponseField>
    <ResponseField name="created_time" type="integer">Unix timestamp in seconds</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable>
    <ResponseField name="limit" type="integer">Requested limit</ResponseField>
    <ResponseField name="offset" type="integer">Current offset</ResponseField>
    <ResponseField name="total" type="integer">Total matching trades</ResponseField>
    <ResponseField name="has_more" type="boolean">Whether more results are available</ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
# Trades for a specific market
curl "https://api.predexon.com/v1/kalshi/trades?ticker=KXBTC-25JAN10-B100000&limit=50"

# Trades for all markets in an event
curl "https://api.predexon.com/v1/kalshi/trades?event_ticker=KXBTC-25JAN10&limit=100"
```

```python Python
import requests

# Get recent trades for a market
response = requests.get(
    "https://api.predexon.com/v1/kalshi/trades",
    params={
        "ticker": "KXBTC-25JAN10-B100000",
        "order": "desc",
        "limit": 50
    }
)

for trade in response.json()["trades"]:
```
