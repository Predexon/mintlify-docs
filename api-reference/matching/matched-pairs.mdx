## Overview

Get all active exact-matched market pairs between Polymarket and Kalshi. This endpoint returns pre-computed pairs where both markets are currently active, ordered by similarity score (highest first).

<ParamField query="min_similarity" type="integer">
  Minimum similarity score filter (0-100). Only return pairs with similarity >= this value.
</ParamField>

<ParamField query="limit" type="integer" default="50">
  Maximum number of pairs to return (1-50)
</ParamField>

<ParamField query="pagination_key" type="string">
  Pagination key for cursor-based pagination (from previous response)
</ParamField>

<ResponseField name="pairs" type="array">
  <Expandable title="Matched pair">
    <ResponseField name="POLYMARKET" type="object">
      <Expandable>
        <ResponseField name="market_slug" type="string">Polymarket market slug</ResponseField>
        <ResponseField name="title" type="string">Market title</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="KALSHI" type="object">
      <Expandable>
        <ResponseField name="market_ticker" type="string">Kalshi market ticker</ResponseField>
        <ResponseField name="title" type="string">Market title</ResponseField>
        <ResponseField name="yes_subtitle" type="string">Yes outcome subtitle</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="similarity" type="integer">
      Similarity score (0-100). May be `null` for older matches.
    </ResponseField>
    <ResponseField name="explanation" type="string">
      Why these markets are considered an exact match. May be `null` for older matches.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable>
    <ResponseField name="limit" type="integer">Requested limit</ResponseField>
    <ResponseField name="count" type="integer">Number of pairs returned</ResponseField>
    <ResponseField name="pagination_key" type="string">Key for next page (null if no more)</ResponseField>
    <ResponseField name="has_more" type="boolean">Whether more results are available</ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
# Get top matched pairs
curl "https://api.predexon.com/v1/matching-markets/pairs?limit=10"

# Filter by minimum similarity
curl "https://api.predexon.com/v1/matching-markets/pairs?min_similarity=95&limit=20"
```

```python Python
import requests

# Get high-confidence matched pairs
response = requests.get(
    "https://api.predexon.com/v1/matching-markets/pairs",
    params={
        "min_similarity": 95,
        "limit": 20
    }
)

data = response.json()
print(f"Found {len(data['pairs'])} pairs")

for pair in data["pairs"]:
    poly = pair["POLYMARKET"]
    kalshi = pair["KALSHI"]
    print(f"\nSimilarity: {pair['similarity']}")
    print(f"  Polymarket: {poly['title']}")
    print(f"  Kalshi: {kalshi['title']}")
```

```typescript TypeScript
const response = await fetch(
  "https://api.predexon.com/v1/matching-markets/pairs?min_similarity=95&limit=20"
);
const { pairs, pagination } = await response.json();

pairs.forEach(pair => {
  console.log(`${pair.similarity}: ${pair.POLYMARKET.title} <=> ${pair.KALSHI.title}`);
});
```
